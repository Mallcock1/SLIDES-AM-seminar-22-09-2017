\begin{frame}
\frametitle{Governing equations}
Ideal MHD equations: \hspace{3.8cm} Conservation of:
\begin{align*}
\rho\frac{\textrm{D}\boldsymbol{v}}{\textrm{D}{t}}&=-\nabla{p}-\alert<2>{\frac{1}{\mu}\boldsymbol{B}\times(\nabla\times\boldsymbol{B})}, & \text{momentum} \\
\frac{\partial\rho}{\partial{t}}+\nabla\cdot(\rho\boldsymbol{v})&=0, & \text{mass} \\
\frac{\textrm{D}}{\textrm{D}{t}}\left(\frac{p}{\rho^\gamma}\right)&=0, & \text{energy} \\
\alert<2>{\frac{\partial{\boldsymbol{B}}}{\partial{t}}}& = \alert<2>{\nabla\times(\boldsymbol{v}\times\boldsymbol{B})}, & \alert<2>{\text{magnetic flux}}
\end{align*}

\small
\centering
\vspace{0.7cm}
\begin{tabular}{ll}
$\boldsymbol{v}=$ plasma velocity, & $\alert<2>{\boldsymbol{B}}=$ \alert<2>{magnetic field strength},
\\
$\rho=$~density, & $p=$~pressure,
\\
$\alert<2>{\mu}=$ \alert<2>{magnetic permeability}, & $\gamma=$ adiabatic index.
\end{tabular}
\end{frame}


\begin{frame}
\frametitle{Asymmetric slab modes}

\begin{block}{Dispersion relation:}
\vspace{-0.5cm}
\begin{align*}
&\frac{\omega^4{m_0}^2}{k^2{v_A}^2-\omega^2}+\frac{\rho_0}{\alert<2>{\rho_1}}\alert<2>{m_1}\frac{\rho_0}{\alert<2>{\rho_2}}\alert<2>{m_2}(k^2{v_A}^2-\omega^2) \\
&-\frac{1}{2}{m_0}\omega^2\left(\frac{\rho_0}{\alert<2>{\rho_1}}\alert<2>{m_1}+\frac{\rho_0}{\alert<2>{\rho_2}}\alert<2>{m_2}\right)\left(\tanh{{m_0}x_0}+\coth{{m_0}x_0}\right)=0,
\end{align*}
\end{block}

\scriptsize
\begin{equation*}
{m_0}^2=\frac{(k^2{v_A}^2-\omega^2)(k^2c_0^2-\omega^2)}{(c_0^2+{v_A}^2)(k^2{c_T}^2-\omega^2)}, \qquad
\alert<2>{m_{1,2}}^2=k^2-\frac{\omega^2}{\alert<2>{c_{1,2}}^2},
\end{equation*}
\begin{equation*}
{c_T}^2=\frac{c_0^2{v_A}^2}{c_0^2+{v_A}^2}, \qquad {v_A}=\frac{{B_0}}{\sqrt{{\mu}\rho_0}},
\end{equation*}
See \textbf{Allcock} and Erd\'{e}lyi, 2017.
\end{frame}

\vspace{-0.3cm}
\begin{equation*}
\Delta_\mathrm{min} = \frac{1}{m_0}\tanh^{-1}(D)
\hspace{2cm}
\Delta_\mathrm{min} = \frac{1}{m_0}\tanh^{-1}\left( \frac{1}{D} \right)
\end{equation*}
\end{block}

\begin{equation*}
\text{where} \quad
D = \frac{(k^2{v_A}^2-\omega^2)m_2\frac{\rho_0}{\rho_2}\tanh({m_0}x_0) - \omega^2{m_0}}{(k^2{v_A}^2-\omega^2)m_2\frac{\rho_0}{\rho_2} - \omega^2{m_0}\tanh({m_0}x_0)}
\end{equation*}
